305  public function getItemByDirection($uploadId, $itemId, $direction, $options)
306  {
307  $uploadTreeTableName = $this->getUploadtreeTableName($uploadId);
308  $originItem = $this->getUploadEntry($itemId, $uploadTreeTableName);
309  $originLft = $originItem['lft'];
310 
311  $options[UploadTreeProxy::OPT_ITEM_FILTER] = " AND ut.ufile_mode & (3<<28) = 0";
312  $uploadTreeViewName = 'items2care';
313 
314  if ($direction == self::DIR_FWD) {
315  $uploadTreeViewName .= 'fwd';
316  $options[UploadTreeProxy::OPT_ITEM_FILTER] .= " AND lft>$1";
317  $order = 'ASC';
318  } else {
319  $uploadTreeViewName .= 'bwd';
320  $options[UploadTreeProxy::OPT_ITEM_FILTER] .= " AND lft<$1";
321  $order = 'DESC';
322  }
323 
324  $uploadTreeView = new UploadTreeProxy($uploadId, $options, $uploadTreeTableName, $uploadTreeViewName);
325  $statementName = __METHOD__ . ".$uploadTreeViewName.";
326  $query = $uploadTreeView->getDbViewQuery()." ORDER BY lft $order";
327 
328  $newItemRow = $this->dbManager->getSingleRow("$query LIMIT 1", array($originLft), $statementName);
329  if ($newItemRow) {
330  return $this->createItem($newItemRow, $uploadTreeTableName);
331  } else {
332  return self::NOT_FOUND;
333  }
334  }

568  public function getPFileDataPerFileName(ItemTreeBounds $itemTreeBounds)
569  {
570  $uploadTreeTableName = $itemTreeBounds->getUploadTreeTableName();
571  $statementName = __METHOD__ . '.' . $uploadTreeTableName;
572  $param = array();
573 
574  $param[] = $itemTreeBounds->getLeft();
575  $param[] = $itemTreeBounds->getRight();
576  $condition = " lft BETWEEN $1 AND $2";
577  $condition .= " AND (ufile_mode & (1<<28)) = 0";
578 
579  if ('uploadtree_a' == $uploadTreeTableName) {
580  $param[] = $itemTreeBounds->getUploadId();
581  $condition .= " AND upload_fk=$".count($param);
582  }
583 
584  $sql = "
585 SELECT ufile_name, uploadtree_pk, lft, rgt, ufile_mode,
586  pfile_pk, pfile_md5, pfile_sha1, pfile_sha256
587 FROM $uploadTreeTableName
588  LEFT JOIN pfile
589  ON pfile_fk = pfile_pk
590 WHERE $condition
591 ORDER BY lft asc
592 ";
593 
594  $this->dbManager->prepare($statementName, $sql);
595  $result = $this->dbManager->execute($statementName, $param);
596  $pfilePerFileName = array();
597 
598  $row = $this->dbManager->fetchArray($result);
599  $pathStack = array($row['ufile_name']);
600  $rgtStack = array($row['rgt']);
601  $lastLft = $row['lft'];
602  $this->addToPFilePerFileName($pfilePerFileName, $pathStack, $row);
603  while ($row = $this->dbManager->fetchArray($result)) {
604  if ($row['lft'] < $lastLft) {
605  continue;
606  }
607 
608  $this->updateStackState($pathStack, $rgtStack, $lastLft, $row);
609  $this->addToPFilePerFileName($pfilePerFileName, $pathStack, $row);
610  }
611  $this->dbManager->freeResult($result);
612  return $pfilePerFileName;
613  }

615  private function updateStackState(&$pathStack, &$rgtStack, &$lastLft, $row)
616  {
617  if ($row['lft'] >= $lastLft) {
618  while (count($rgtStack) > 0 && $row['lft'] > $rgtStack[count($rgtStack)-1]) {
619  array_pop($pathStack);
620  array_pop($rgtStack);
621  }
622  if ($row['lft'] > $lastLft) {
623  array_push($pathStack, $row['ufile_name']);
624  array_push($rgtStack, $row['rgt']);
625  $lastLft = $row['lft'];
626  }
627  }
628  }
