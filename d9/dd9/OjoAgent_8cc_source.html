64  scanString(fileContent, regLicenseList, licenseList, 0, false);

121 void OjoAgent::scanString(const string &text, boost::regex reg,
122  vector<ojomatch> &result, unsigned int offset, bool isDualTest)
123 {
124  string::const_iterator end = text.end();
125  string::const_iterator pos = text.begin();
126 
127  while (pos != end)
128  {
129  // Find next match
130  boost::smatch res;
131  if (boost::regex_search(pos, end, res, reg))
132  {
133  string content = "Dual-license";
134  if (! isDualTest)
135  {
136  content = res[1].str();
137  }
138  // Found match
139  result.push_back(
140  ojomatch(offset + res.position(1),
141  offset + res.position(1) + res.length(1),
142  res.length(1),
143  content));
144  pos = res[0].second;
145  offset += res.position() + res.length();
146  }
147  else
148  {
149  // No match found
150  break;
151  }
152  }
153 }
