57  function getAgentFileLicenseMatches(ItemTreeBounds $itemTreeBounds, $usageId=LicenseMap::TRIVIAL)
58  {
59  $uploadTreeTableName = $itemTreeBounds->getUploadTreeTableName();
60  $statementName = __METHOD__ . ".$uploadTreeTableName.$usageId";
61  $params = array($itemTreeBounds->getUploadId(), $itemTreeBounds->getLeft(), $itemTreeBounds->getRight());
62  if ($usageId==LicenseMap::TRIVIAL) {
63  $licenseJoin = "license_ref mlr ON license_file.rf_fk = mlr.rf_pk";
64  } else {
65  $params[] = $usageId;
66  $licenseMapCte = LicenseMap::getMappedLicenseRefView('$4');
67  $licenseJoin = "($licenseMapCte) AS mlr ON license_file.rf_fk = mlr.rf_origin";
68  }
69 
70  $this->dbManager->prepare($statementName,
71  "SELECT LFR.rf_shortname AS license_shortname,
72  LFR.rf_fullname AS license_fullname,
73  LFR.rf_pk AS license_id,
74  LFR.fl_pk AS license_file_id,
75  LFR.pfile_fk as file_id,
76  LFR.rf_match_pct AS percent_match,
77  AG.agent_name AS agent_name,
78  AG.agent_pk AS agent_id,
79  AG.agent_rev AS agent_revision
80  FROM ( SELECT mlr.rf_fullname, mlr.rf_shortname, mlr.rf_pk, license_file.fl_pk, license_file.agent_fk, license_file.pfile_fk, license_file.rf_match_pct
81  FROM license_file JOIN $licenseJoin) as LFR
82  INNER JOIN $uploadTreeTableName as UT ON UT.pfile_fk = LFR.pfile_fk
83  INNER JOIN agent as AG ON AG.agent_pk = LFR.agent_fk
84  WHERE AG.agent_enabled='true' and
85  UT.upload_fk=$1 AND UT.lft BETWEEN $2 and $3
86  ORDER BY license_shortname ASC, percent_match DESC");
87  $result = $this->dbManager->execute($statementName, $params);
88  $matches = array();
89  while ($row = $this->dbManager->fetchArray($result)) {
90  $licenseRef = new LicenseRef(intval($row['license_id']), $row['license_shortname'], $row['license_fullname']);
91  $agentRef = new AgentRef(intval($row['agent_id']), $row['agent_name'], $row['agent_revision']);
92  $matches[] = new LicenseMatch(intval($row['file_id']), $licenseRef, $agentRef, intval($row['license_file_id']), intval($row['percent_match']));
93  }
94 
95  $this->dbManager->freeResult($result);
96  return $matches;

303  public function getLicensesPerFileNameForAgentId(ItemTreeBounds $itemTreeBounds,
304  $selectedAgentIds=null,
305  $includeSubfolders=true,
306  $excluding='',
307  $ignore=false,
308  &$clearingDecisionsForLicList = array())
309  {
310  $uploadTreeTableName = $itemTreeBounds->getUploadTreeTableName();
311  $statementName = __METHOD__ . '.' . $uploadTreeTableName;
312  $param = array();
313 
314  $condition = " (ufile_mode & (1<<28)) = 0";
315  if ($includeSubfolders) {
316  $param[] = $itemTreeBounds->getLeft();
317  $param[] = $itemTreeBounds->getRight();
318  $condition .= " AND lft BETWEEN $1 AND $2";
319  $statementName .= ".subfolders";
320  } else {
321  $param[] = $itemTreeBounds->getItemId();
322  $condition .= " AND realparent = $1";
323  }
324 
325  if ('uploadtree_a' == $uploadTreeTableName) {
326  $param[] = $itemTreeBounds->getUploadId();
327  $condition .= " AND upload_fk=$".count($param);
328  }
329 
330  $agentSelect = "";
331  if ($selectedAgentIds !== null) {
332  $statementName .= ".".count($selectedAgentIds)."agents";
333  $agentSelect = "WHERE agent_fk IS NULL";
334  foreach ($selectedAgentIds as $selectedAgentId) {
335  $param[] = $selectedAgentId;
336  $agentSelect .= " OR agent_fk = $".count($param);
337  }
338  }
339 
340  $sql = "
341 SELECT uploadtree_pk, ufile_name, lft, rgt, ufile_mode,
342  rf_shortname, agent_fk
343 FROM (SELECT
344  uploadtree_pk, ufile_name,
345  lft, rgt, ufile_mode, pfile_fk
346  FROM $uploadTreeTableName
347  WHERE $condition) AS subselect1
348 LEFT JOIN (SELECT rf_shortname,pfile_fk,agent_fk
349  FROM license_file, license_ref
350  WHERE rf_fk = rf_pk) AS subselect2
351  ON subselect1.pfile_fk = subselect2.pfile_fk
352 $agentSelect
353 ORDER BY lft asc
354 ";
355 
356  $this->dbManager->prepare($statementName, $sql);
357  $result = $this->dbManager->execute($statementName, $param);
358  $licensesPerFileName = array();
359 
360  $row = $this->dbManager->fetchArray($result);
361  $pathStack = array($row['ufile_name']);
362  $rgtStack = array($row['rgt']);
363  $lastLft = $row['lft'];
364  $path = implode($pathStack,'/');
365  $this->addToLicensesPerFileName($licensesPerFileName, $path, $row, $ignore, $clearingDecisionsForLicList);
366  while ($row = $this->dbManager->fetchArray($result)) {
367  if (!empty($excluding) && false!==strpos("/$row[ufile_name]/", $excluding)) {
368  $lastLft = $row['rgt'] + 1;
369  continue;
370  }
371  if ($row['lft'] < $lastLft) {
372  continue;
373  }
374 
375  $this->updateStackState($pathStack, $rgtStack, $lastLft, $row);
376  $path = implode($pathStack,'/');
377  $this->addToLicensesPerFileName($licensesPerFileName, $path, $row, $ignore, $clearingDecisionsForLicList);
378  }
379  $this->dbManager->freeResult($result);
380  return array_reverse($licensesPerFileName);
381  }

383  private function updateStackState(&$pathStack, &$rgtStack, &$lastLft, $row)
384  {
385  if ($row['lft'] >= $lastLft) {
386  while (count($rgtStack) > 0 && $row['lft'] > $rgtStack[count($rgtStack)-1]) {
387  array_pop($pathStack);
388  array_pop($rgtStack);
389  }
390  if ($row['lft'] > $lastLft) {
391  array_push($pathStack, $row['ufile_name']);
392  array_push($rgtStack, $row['rgt']);
393  $lastLft = $row['lft'];
394  }
395  }
396  }

398  private function addToLicensesPerFileName(&$licensesPerFileName, $path, $row, $ignore, &$clearingDecisionsForLicList = array())
399  {
400  if (($row['ufile_mode'] & (1 << 29)) == 0) {
401  if ($row['rf_shortname']) {
402  $licensesPerFileName[$path]['scanResults'][] = $row['rf_shortname'];
403  if (array_key_exists($row['uploadtree_pk'], $clearingDecisionsForLicList)) {
404  $licensesPerFileName[$path]['concludedResults'][] = $clearingDecisionsForLicList[$row['uploadtree_pk']];
405  }
406  }
407  } else if (!$ignore) {
408  $licensesPerFileName[$path] = false;
409  }
410  }
