135  function processUploadId($uploadId)
136  {

139  $pfileFks = $this->softwareHeritageDao->getSoftwareHeritagePfileFk($uploadId);
140  $agentId = $this->agentDao->getCurrentAgentId("softwareHeritage");
141  $maxTime = $this->configuration['maxtime'];
142  $maxTime = ($maxTime < 2) ? 2 : $maxTime;
143  foreach ($pfileFileDetails as $pfileDetail) {
144  if (!in_array($pfileDetail['pfile_pk'], array_column($pfileFks, 'pfile_fk'))) {
145  $this->processEachPfileForSWH($pfileDetail, $agentId, $maxTime);
146  }
147  $this->heartbeat(1);
148  }
149  return true;

159  function processEachPfileForSWH($pfileDetail, $agentId, $maxTime)
160  {
161  list ($currentStatus, $currentResult) = $this->getSoftwareHeritageLicense($pfileDetail['sha256']);
162  if (SoftwareHeritageDao::SWH_RATELIMIT_EXCEED == $currentStatus) {
163  $this->heartbeat(0); //Fake heartbeat to keep the agent alive.
164  $timeToReset = $currentResult - time();
165  print "INFO :Software Heritage X-RateLimit-Limit reached. Next slot unlocks in ".gmdate("H:i:s", $timeToReset)."\n";
166  if ($timeToReset > $maxTime) {
167  sleep($maxTime);
168  } else {
169  sleep($timeToReset);
170  }
171  $this->processEachPfileForSWH($pfileDetail, $agentId, $maxTime);
172  } else {
173  $this->insertSoftwareHeritageRecord($pfileDetail['pfile_pk'], $currentResult, $agentId, $currentStatus);
174  }
175 
176  return true;
177  }

185  protected function getSoftwareHeritageLicense($sha256)
186  {
187  $sha256 = strtolower($sha256);
188  $URIToGetContent = $this->configuration['uri'] . $sha256;
189  $URIToGetLicenses = $URIToGetContent . $this->configuration['content'];
190 
191  try {
192  $response = $this->guzzleClient->get($URIToGetLicenses);
193  $statusCode = $response->getStatusCode();
194  $cookedResult = array();
195  if ($statusCode == SoftwareHeritageDao::SWH_STATUS_OK) {
196  $responseContent = json_decode($response->getBody()->getContents(),true);
197  $cookedResult = $responseContent["facts"][0]["licenses"];
198  } else if ($statusCode == SoftwareHeritageDao::SWH_RATELIMIT_EXCEED) {
199  $responseContent = $response->getHeaders();
200  $cookedResult = $responseContent["X-RateLimit-Reset"][0];
201  } else if ($statusCode == SoftwareHeritageDao::SWH_NOT_FOUND) {
202  $response = $this->guzzleClient->get($URIToGetContent);
203  $responseContent = json_decode($response->getBody(),true);
204  if (isset($responseContent["status"])) {
205  $statusCode = SoftwareHeritageDao::SWH_STATUS_OK;
206  }
207  }
208  return array($statusCode, $cookedResult);
209  } catch (RequestException $e) {
210  echo "Sorry, something went wrong. check if the host is accessible!\n";
211  echo Psr7\str($e->getRequest());
212  if ($e->hasResponse()) {
213  echo Psr7\str($e->getResponse());
214  }
215  $this->scheduler_disconnect(1);
216  exit;
217  }
218  }
