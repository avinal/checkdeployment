71 char *unescapePathSeparator(const char* json)
72 {
73  const char *escapedSeparator = "\\/";
74  const char *pathSeparator = "/";
75  const int escPathLen = 2;
76  const int pathSepLen = 1;
77  size_t resultLength = 0;
78  size_t remainingLength = -1;
79  char *result;
80  char *tmp;
81  char *tempjson;
82  int count;
83  if (!json)
84  {
85  return NULL;
86  }
87  tempjson = strdup(json);
88 
89  tmp = tempjson;
90  for (count = 0; (tmp = strstr(tmp, escapedSeparator)); count++)
91  {
92  tmp += escPathLen;
93  }
94 
95  resultLength = strlen(tempjson) - ((escPathLen - pathSepLen) * count);
96 
97  result = (char*) calloc(resultLength + 1, sizeof(char));
98 
99  strncpy(result, strtok(tempjson, escapedSeparator), resultLength);
100  remainingLength = resultLength - strlen(result);
101 
102  while (count-- && remainingLength > 0)
103  {
104  strncat(result, pathSeparator, remainingLength);
105  strncat(result, strtok(NULL, escapedSeparator), remainingLength - 1);
106  remainingLength = resultLength - strlen(result);
107  }
108  free(tempjson);
109  return result;
110 }
