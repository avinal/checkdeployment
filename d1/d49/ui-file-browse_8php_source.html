117  protected function handle(Request $request)
118  {
119  $upload = intval($request->get("upload"));
120  $groupId = Auth::getGroupId();
121  if (!$this->uploadDao->isAccessible($upload, $groupId)) {
122  return $this->flushContent(_("Permission Denied"));
123  }
124 
125  $item = intval($request->get("item"));
126 
127  $vars['baseuri'] = Traceback_uri();
128  $vars['uploadId'] = $upload;
129  $this->uploadtree_tablename = $this->uploadDao->getUploadtreeTableName($upload);
130  if ($request->get('show')=='quick') {
131  $item = $this->uploadDao->getFatItemId($item,$upload,$this->uploadtree_tablename);
132  }
133  $vars['itemId'] = $item;
134 
135  $itemTreeBounds = $this->uploadDao->getItemTreeBounds($item, $this->uploadtree_tablename);
136  $left = $itemTreeBounds->getLeft();
137  if (empty($left)) {
138  return $this->flushContent(_("Job unpack/adj2nest hasn't completed."));
139  }
140  $histVars = $this->showUploadHist($itemTreeBounds);
141  if (is_a($histVars, 'Symfony\\Component\\HttpFoundation\\RedirectResponse')) {
142  return $histVars;
143  }
144  $vars = array_merge($vars, $histVars);
145 
146  $vars['micromenu'] = Dir2Browse($this->Name, $item, NULL, $showBox = 0, "Browse", -1, '', '', $this->uploadtree_tablename);
147 
148  $allLicensesPre = $this->licenseDao->getLicenseArray();
149  $allLicenses = array();
150  foreach ($allLicensesPre as $value) {
151  $allLicenses[$value['shortname']] = array('rf_pk' => $value['id']);
152  }
153  $vars['scannerLicenses'] = $allLicenses;
154 
155  $vars['content'] = js_url();
156 
157  return $this->render("file-browse.html.twig",$this->mergeWithDefault($vars));
158  }

166  private function showUploadHist(ItemTreeBounds $itemTreeBounds)
167  {
168  $groupId = Auth::getGroupId();
169  $selectedAgentId = GetParm('agentId', PARM_INTEGER);
170  $tag_pk = GetParm("tag", PARM_INTEGER);
171 
172  $uploadId = $itemTreeBounds->getUploadId();
173  $scannerAgents = array_keys($this->agentNames);
174  $scanJobProxy = new ScanJobProxy($this->agentDao, $uploadId);
175  $scannerVars = $scanJobProxy->createAgentStatus($scannerAgents);
176  $agentMap = $scanJobProxy->getAgentMap();
177 
178  $vars = array('agentId' => $selectedAgentId,
179  'agentMap' => $agentMap,
180  'scanners'=>$scannerVars);
181 
182  $selectedAgentIds = empty($selectedAgentId) ? $scanJobProxy->getLatestSuccessfulAgentIds() : $selectedAgentId;
183 
184  $this->licenseProjector = new LicenseMap($this->getObject('db.manager'),$groupId,LicenseMap::CONCLUSION,true);
185  $dirVars = $this->countFileListing($itemTreeBounds);
186  $childCount = $dirVars['iTotalRecords'];
187  /***************************************
188  * Problem: $ChildCount can be zero if you have a container that does not
189  * unpack to a directory. For example:
190  * file.gz extracts to archive.txt that contains a license.
191  * Same problem seen with .pdf and .Z files.
192  * Solution: if $ChildCount == 0, then just view the license!
193  *
194  * $ChildCount can also be zero if the directory is empty.
195  * **************************************/
196  if ($childCount == 0) {
197  return new RedirectResponse("?mod=view-license" . Traceback_parm_keep(array("upload", "item")));
198  }
199 
200  $vars['licenseUri'] = Traceback_uri() . "?mod=popup-license&rf=";
201  $vars['bulkUri'] = Traceback_uri() . "?mod=popup-license";
202 
203  return array_merge($vars, $dirVars);
204  }

216  $vars['fileSwitch'] = $isFlat ? $uri : $uri."&flatten=yes";
217  return $vars;
