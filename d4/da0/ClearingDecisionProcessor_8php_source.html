81  public function hasUnhandledScannerDetectedLicenses(ItemTreeBounds $itemTreeBounds, $groupId, $additionalEventIds = array(), $licenseMap=null)
82  {
83  if (!empty($licenseMap) && !($licenseMap instanceof LicenseMap)) {
84  throw new Exception('invalid license map');
85  }
86  $userEvents = $this->clearingDao->getRelevantClearingEvents($itemTreeBounds, $groupId);
87  $usageId = empty($licenseMap) ? LicenseMap::TRIVIAL : $licenseMap->getUsage();
88  $scannerDetectedEvents = $this->agentLicenseEventProcessor->getScannerEvents($itemTreeBounds,$usageId);
89  $eventLicenceIds = array();
90  foreach (array_keys($userEvents) as $licenseId) {
91  $eventLicenceIds[empty($licenseMap)? $licenseId: $licenseMap->getProjectedId($licenseId)] = $licenseId;
92  }
93  foreach (array_keys($additionalEventIds) as $licenseId) {
94  $eventLicenceIds[empty($licenseMap)? $licenseId: $licenseMap->getProjectedId($licenseId)] = $licenseId;
95  }
96  foreach (array_keys($scannerDetectedEvents) as $licenseId) {
97  if (!array_key_exists(empty($licenseMap)? $licenseId: $licenseMap->getProjectedId($licenseId), $eventLicenceIds)) {
98  return true;
99  }
100  }
101  return false;
102  }

114  private function insertClearingEventsForAgentFindings(ItemTreeBounds $itemBounds, $userId, $groupId, $remove = false, $type = ClearingEventTypes::AGENT, $removedIds = array())
115  {
116  $eventIds = array();
117  foreach ($this->agentLicenseEventProcessor->getScannerEvents($itemBounds) as $licenseId => $scannerEvents) {
118  if (array_key_exists($licenseId, $removedIds)) {
119  continue;

121  $scannerLicenseRef = $scannerEvents[0]->getLicenseRef();
122  $eventIds[$scannerLicenseRef->getId()] = $this->clearingDao->insertClearingEvent($itemBounds->getItemId(), $userId, $groupId, $scannerLicenseRef->getId(), $remove, $type);
123  }
124  return $eventIds;

135  private function clearingDecisionIsDifferentFrom(ClearingDecision $decision, $type, $scope, $clearingEventIds)
136  {
137  $clearingEvents = $decision->getClearingEvents();
138  if (count($clearingEvents) != count($clearingEventIds)) {
139  return true;
140  }
141 
142  foreach ($clearingEvents as $clearingEvent) {
143  if (false === array_search($clearingEvent->getEventId(), $clearingEventIds)) {
144  return true;
145  }
146  }
147  return ($type !== $decision->getType()) || ($scope !== $decision->getScope());
148  }

159  public function makeDecisionFromLastEvents(ItemTreeBounds $itemBounds, $userId, $groupId, $type, $global, $additionalEventIds = array())
160  {
161  if ($type < self::NO_LICENSE_KNOWN_DECISION_TYPE) {
162  return;
163  }
164  $this->dbManager->begin();
165 
166  $itemId = $itemBounds->getItemId();
167  $includeSubFolders = false;
168  if (($global == DecisionScopes::REPO) && ($type != self::NO_LICENSE_KNOWN_DECISION_TYPE)) {
169  $includeSubFolders = true;
170  }
171  $previousEvents = $this->clearingDao->getRelevantClearingEvents($itemBounds, $groupId, $includeSubFolders);
172  if ($type === self::NO_LICENSE_KNOWN_DECISION_TYPE) {
173  $type = DecisionTypes::IDENTIFIED;
174  $clearingEventIds = $this->insertClearingEventsForAgentFindings($itemBounds, $userId, $groupId, true, ClearingEventTypes::USER);
175  foreach ($previousEvents as $eventId => $clearingEvent) {
176  if (!in_array($eventId, $clearingEventIds) && !$clearingEvent->isRemoved()) {
177  $licenseId = $clearingEvent->getLicenseId();
178  $newEventId = $this->clearingDao->insertClearingEvent($itemBounds->getItemId(), $userId, $groupId, $licenseId, true);
179  $clearingEventIds[$licenseId] = $newEventId;
180  }
181  }
182  } else {
183  $clearingEventIds = $this->insertClearingEventsForAgentFindings($itemBounds, $userId, $groupId, false, ClearingEventTypes::AGENT, $previousEvents);
184  foreach ($previousEvents as $clearingEvent) {
185  $clearingEventIds[$clearingEvent->getLicenseId()] = $clearingEvent->getEventId();
186  }
187  }
188 
189  $currentDecision = $this->clearingDao->getRelevantClearingDecision($itemBounds, $groupId);
190  $clearingEventIds = array_unique(array_merge($clearingEventIds, $additionalEventIds));
191 
192  $scope = $global ? DecisionScopes::REPO : DecisionScopes::ITEM;
193  if (null === $currentDecision || $this->clearingDecisionIsDifferentFrom($currentDecision, $type, $scope, $clearingEventIds)) {
194  $this->clearingDao->createDecisionFromEvents($itemBounds->getItemId(), $userId, $groupId, $type, $scope,
195  $clearingEventIds);
196  } else {
197  $this->clearingDao->removeWipClearingDecision($itemId, $groupId);
198  }
199 
200  $this->dbManager->commit();
201  }

211  public function getCurrentClearings(ItemTreeBounds $itemTreeBounds, $groupId, $usageId=LicenseMap::TRIVIAL)
212  {
213  $agentEvents = $this->agentLicenseEventProcessor->getScannerEvents($itemTreeBounds, $usageId);
214  $events = $this->clearingDao->getRelevantClearingEvents($itemTreeBounds, $groupId);
215 
216  $addedResults = array();
217  $removedResults = array();
218 
219  foreach (array_unique(array_merge(array_keys($events), array_keys($agentEvents))) as $licenseId) {
220  $licenseDecisionEvent = array_key_exists($licenseId, $events) ? $events[$licenseId] : null;
221  $agentClearingEvents = array_key_exists($licenseId, $agentEvents) ? $agentEvents[$licenseId] : array();
222 
223  if (($licenseDecisionEvent === null) && (count($agentClearingEvents) == 0)) {
224  throw new Exception('not in merge');
225  }
226  $licenseDecisionResult = new ClearingResult($licenseDecisionEvent, $agentClearingEvents);
227  if ($licenseDecisionResult->isRemoved()) {
228  $removedResults[$licenseId] = $licenseDecisionResult;
229  } else {
230  $addedResults[$licenseId] = $licenseDecisionResult;
231  }
232  }
233 
234  return array($addedResults, $removedResults);
235  }
