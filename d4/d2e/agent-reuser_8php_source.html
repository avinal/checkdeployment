99  public function scheduleAgent($jobId, $uploadId, &$errorMsg, $request)
100  {
101  $reuseUploadPair = explode(',',
102  $request->get(self::UPLOAD_TO_REUSE_SELECTOR_NAME), 2);
103  if (count($reuseUploadPair) == 2) {
104  list($reuseUploadId, $reuseGroupId) = $reuseUploadPair;
105  } else {
106  $errorMsg .= 'no reuse upload id given';
107  return - 1;
108  }
109  $groupId = $request->get('groupId', Auth::getGroupId());
110 
111  $getReuseValue = $request->get('reuseMode') ?: array();
112  $reuserDependencies = array("agent_adj2nest");
113 
114  $reuseMode = UploadDao::REUSE_NONE;
115  foreach ($getReuseValue as $currentReuseValue) {
116  switch ($currentReuseValue) {
117  case 'reuseMain':
118  $reuseMode |= UploadDao::REUSE_MAIN;
119  break;
120  case 'reuseEnhanced':
121  $reuseMode |= UploadDao::REUSE_ENHANCED;
122  break;
123  case 'reuseConf':
124  $reuseMode |= UploadDao::REUSE_CONF;
125  break;
126  case 'reuseCopyright':
127  $reuseMode |= UploadDao::REUSE_COPYRIGHT;
128  break;
129  }
130  }
131 
132  $reuserDependencies = array_merge($reuserDependencies,
133  $this->getReuserDependencies($request));
134 
135  $this->createPackageLink($uploadId, $reuseUploadId, $groupId, $reuseGroupId,
136  $reuseMode);
137 
138  return $this->doAgentAdd($jobId, $uploadId, $errorMsg,
139  array_unique($reuserDependencies), $uploadId);
140  }

150  protected function createPackageLink($uploadId, $reuseUploadId, $groupId, $reuseGroupId, $reuseMode=0)
151  {
152  /* @var $packageDao PackageDao */
153  $packageDao = $GLOBALS['container']->get('dao.package');
154  $newUpload = $this->uploadDao->getUpload($uploadId);
155  $uploadForReuse = $this->uploadDao->getUpload($reuseUploadId);
156 
157  $package = $packageDao->findPackageForUpload($reuseUploadId);
158 
159  if ($package === null) {
160  $packageName = StringOperation::getCommonHead($uploadForReuse->getFilename(), $newUpload->getFilename());
161  $package = $packageDao->createPackage($packageName ?: $uploadForReuse->getFilename());
162  $packageDao->addUploadToPackage($reuseUploadId, $package);
163  }
164 
165  $packageDao->addUploadToPackage($uploadId, $package);
166 
167  $this->uploadDao->addReusedUpload($uploadId, $reuseUploadId, $groupId, $reuseGroupId, $reuseMode);
168  }
