85  function Output()
86  {
87  $returnValue = 0;
88  if ($this->State != PLUGIN_STATE_READY) {
89  return 0;
90  }
91 
92  $action = GetParm("action", PARM_STRING);
93  $upload = GetParm("upload", PARM_INTEGER);
94  $type = GetParm("type", PARM_STRING);
95 
96  if ($action=="deletedecision" || $action=="undodecision") {
97  $decision = GetParm("decision", PARM_INTEGER);
98  $pfile = GetParm("pfile", PARM_INTEGER);
99  } else if ($action=="deleteHashDecision" || $action=="undoHashDecision") {
100  $hash = GetParm("hash", PARM_STRING);
101  } else if ($action=="update" || $action=="delete" || $action=="undo") {
102  $id = GetParm("id", PARM_STRING);
103  $getEachID = array_filter(explode(",", trim($id, ',')), function($var) {
104  return $var !== "";
105  });
106  if (count($getEachID) == 4) {
107  list($upload, $item, $hash, $type) = $getEachID;
108  } else {
109  return new Response('bad request while '.$action,
110  Response::HTTP_BAD_REQUEST,
111  array('Content-type'=>'text/plain')
112  );
113  }
114  }
115 
116  /* check upload permissions */
117  if (!(($action == "getData" || $action == "getDeactivatedData") &&
118  ($this->uploadDao->isAccessible($upload, Auth::getGroupId())) ||
119  ($this->uploadDao->isEditable($upload, Auth::getGroupId())))) {
120  $permDeniedText = _("Permission Denied");
121  $returnValue = "<h2>$permDeniedText</h2>";
122  }
123  $this->uploadtree_tablename = $this->uploadDao->getUploadtreeTableName(
124  $upload);
125 
126  if (in_array($type, $this->textFindingTypes) &&
127  ($action == "getData" || $action == "getDeactivatedData")) {
128  $textFindingsHandler = new TextFindingsAjax($this->uploadtree_tablename);
129  if ($action == "getData") {
130  $returnValue = $textFindingsHandler->doGetData($type, $upload);
131  } elseif ($action == "getDeactivatedData") {
132  $returnValue = $textFindingsHandler->doGetData($type, $upload, false);
133  }
134  } else {
135  switch ($action) {
136  case "getData":
137  $returnValue = $this->doGetData($upload);
138  break;
139  case "getDeactivatedData":
140  $returnValue = $this->doGetData($upload, false);
141  break;
142  case "update":
143  $returnValue = $this->doUpdate($item, $hash, $type);
144  break;
145  case "delete":
146  $returnValue = $this->doDelete($item, $hash, $type);
147  break;
148  case "undo":
149  $returnValue = $this->doUndo($item, $hash, $type);
150  break;
151  default:
152  $returnValue = "<h2>" . _("Unknown action") . "</h2>";
153  }
154  }
155  return $returnValue;
156  }

231  protected function getCopyrights($upload_pk, $item, $uploadTreeTableName,
232  $agentId, $type, $filter, $activated = true)
233  {
234  $offset = GetParm('iDisplayStart', PARM_INTEGER);
235  $limit = GetParm('iDisplayLength', PARM_INTEGER);
236 
237  $tableName = $this->getTableName($type);
238  $orderString = $this->getOrderString();
239 
240  list($left, $right) = $this->uploadDao->getLeftAndRight($item,
241  $uploadTreeTableName);
242 
243  if ($filter == "") {
244  $filter = "none";
245  }
246 
247  $sql_upload = "";
248  if ('uploadtree_a' == $uploadTreeTableName) {
249  $sql_upload = " AND UT.upload_fk=$upload_pk ";
250  }
251 
252  $join = "";
253  $filterQuery = "";
254  if ($type == 'statement' && $filter == "nolic") {
255  $noLicStr = "No_license_found";
256  $voidLicStr = "Void";
257  $join = " INNER JOIN license_file AS LF on cp.pfile_fk=LF.pfile_fk ";
258  $filterQuery = " AND LF.rf_fk IN (SELECT rf_pk FROM license_ref WHERE rf_shortname IN ('$noLicStr','$voidLicStr')) ";
259  } else {
260  // No filter, nothing to do
261  }
262  $params = array($left, $right, $agentId);
263 
264  $filterParms = $params;
265  $searchFilter = $this->addSearchFilter($filterParms);
266  $unorderedQuery = "FROM $tableName AS cp " .
267  "INNER JOIN $uploadTreeTableName AS UT ON cp.pfile_fk = UT.pfile_fk " .
268  $join .
269  "WHERE textfinding!='' " .
270  "AND ( UT.lft BETWEEN $1 AND $2 ) " .
271  "AND cp.agent_fk= $3 " .
272  "AND cp.is_enabled=" . ($activated ? 'true' : 'false') .
273  $sql_upload;
274  $totalFilter = $filterQuery . " " . $searchFilter;
275 
276  $grouping = " GROUP BY hash, textfinding ";
277 
278  $countQuery = "SELECT count(*) FROM (SELECT textfinding AS content, count(*) $unorderedQuery $totalFilter $grouping) as K";
279  $iTotalDisplayRecordsRow = $this->dbManager->getSingleRow($countQuery,
280  $filterParms, __METHOD__.$tableName . ".count" .
281  ($activated ? '' : '_deactivated'));
282  $iTotalDisplayRecords = $iTotalDisplayRecordsRow['count'];
283 
284  $countAllQuery = "SELECT count(*) FROM (SELECT textfinding AS content, count(*) $unorderedQuery$grouping) as K";
285  $iTotalRecordsRow = $this->dbManager->getSingleRow($countAllQuery, $params,
286  __METHOD__,$tableName . "count.all" . ($activated ? '' : '_deactivated'));
287  $iTotalRecords = $iTotalRecordsRow['count'];
288 
289  $range = "";
290  $filterParms[] = $offset;
291  $range .= ' OFFSET $' . count($filterParms);
292  $filterParms[] = $limit;
293  $range .= ' LIMIT $' . count($filterParms);
294 
295  $sql = "SELECT textfinding AS content, hash, count(*) as copyright_count " .
296  $unorderedQuery . $totalFilter . " GROUP BY content, hash " .
297  $orderString . $range;
298  $statement = __METHOD__ . $filter.$tableName . $uploadTreeTableName .
299  ($activated ? '' : '_deactivated');
300  $rows = $this->dbManager->getRows($sql, $filterParms, $statement);
301 
302  return array($rows, $iTotalDisplayRecords, $iTotalRecords);
303  }
