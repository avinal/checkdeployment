55  public function getPfile($sha1 = null, $md5 = null, $sha256 = null,
56  $size = null)
57  {
58  $statement = __METHOD__ . ".getPfileId";
59  $sql = "SELECT * FROM pfile WHERE ";
60  $params = [];
61  $conds = [];
62  if (! empty($sha1)) {
63  $statement .= ".sha1";
64  $params[] = strtoupper($sha1);
65  $conds[] = "pfile_sha1 = $" . count($params);
66  }
67  if (! empty($md5)) {
68  $statement .= ".md5";
69  $params[] = strtoupper($md5);
70  $conds[] = "pfile_md5 = $" . count($params);
71  }
72  if (! empty($sha256)) {
73  $statement .= ".sha256";
74  $params[] = strtoupper($sha256);
75  $conds[] = "pfile_sha256 = $" . count($params);
76  }
77  if (! empty($size)) {
78  $statement .= ".size";
79  $params[] = $size;
80  $conds[] = "pfile_size = $" . count($params);
81  }
82  $sql .= join(" AND ", $conds);
83  $row = $this->dbManager->getSingleRow($sql, $params, $statement);
84  return (! empty($row)) ? $row : null;
85  }

92  public function getScannerFindings($pfileId)
93  {
94  $statement = __METHOD__ . ".getScannerFindings";
95  $sql = "SELECT DISTINCT ON(rf_pk) rf_shortname " .
96  "FROM license_file AS lf " .
97  "INNER JOIN ONLY license_ref AS lr ON lf.rf_fk = lr.rf_pk " .
98  "WHERE lf.pfile_fk = $1;";
99  $params = [$pfileId];
100  $rows = $this->dbManager->getRows($sql, $params, $statement);
101  if (! empty($rows) && array_key_exists('rf_shortname', $rows[0])) {
102  $licenses = array_column($rows, 'rf_shortname');
103  natcasesort($licenses);
104  return array_values($licenses);
105  } else {
106  return [];

119  public function getConclusions($groupId, $pfileId)
120  {
121  if (! $this->haveConclusions($groupId, $pfileId)) {
122  return ["NOASSERTION"];
123  }
124  $statement = __METHOD__ . ".getScannerFindings";
125  $sql = "WITH allDecsPfile AS (
126  SELECT cd.pfile_fk, cd.clearing_decision_pk, cd.decision_type,
127  lr.rf_shortname, ce.removed
128  FROM clearing_decision AS cd
129  INNER JOIN clearing_decision_event AS cde
130  ON cde.clearing_decision_fk = cd.clearing_decision_pk
131  INNER JOIN clearing_event AS ce
132  ON cde.clearing_event_fk = ce.clearing_event_pk
133  INNER JOIN license_ref AS lr
134  ON ce.rf_fk = lr.rf_pk
135  WHERE cd.pfile_fk = $1 AND (cd.group_fk = $2 OR cd.scope = " .
136  DecisionScopes::REPO . ")
137  ORDER BY cd.clearing_decision_pk DESC
138 ),
139 rankedDecs AS (
140  SELECT *, rank() OVER (
141  PARTITION BY pfile_fk, rf_shortname ORDER BY clearing_decision_pk DESC
142  ) rnk FROM allDecsPfile
143 )
144 SELECT * FROM rankedDecs
145 WHERE rnk = 1 AND removed = false AND decision_type = " .
146  DecisionTypes::IDENTIFIED . ";";
147  $params = [$pfileId, $groupId];
148  $rows = $this->dbManager->getRows($sql, $params, $statement);
149  if (! empty($rows) && array_key_exists('rf_shortname', $rows[0])) {
150  $licenses = array_column($rows, 'rf_shortname');
151  natcasesort($licenses);
152  return array_values($licenses);
153  } else {
154  return ["NONE"];
155  }
