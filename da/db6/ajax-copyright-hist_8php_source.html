145  $returnValue = $this->doGetData($upload, false);
146  break;

242  protected function getCopyrights($upload_pk, $item, $uploadTreeTableName, $agentId, $type, $filter, $activated = true)
243  {
244  $offset = GetParm('iDisplayStart', PARM_INTEGER);
245  $limit = GetParm('iDisplayLength', PARM_INTEGER);
246 
247  $tableName = $this->getTableName($type);
248  $tableNameEvent = $tableName.'_event';
249  $orderString = $this->getOrderString();
250 
251  list($left, $right) = $this->uploadDao->getLeftAndRight($item, $uploadTreeTableName);
252 
253  if ($filter == "")
254  {
255  $filter = "none";
256  }
257 
258  $sql_upload = "";
259  if ('uploadtree_a' == $uploadTreeTableName)
260  {
261  $sql_upload = " AND UT.upload_fk=$upload_pk ";
262  }
263 
264  $join = "";
265  $filterQuery = "";
266  if ($type == 'statement' && $filter == "nolic")
267  {
268  $noLicStr = "No_license_found";
269  $voidLicStr = "Void";
270  $join = " INNER JOIN license_file AS LF on cp.pfile_fk=LF.pfile_fk ";
271  $filterQuery = " AND LF.rf_fk IN (SELECT rf_pk FROM license_ref WHERE rf_shortname IN ('$noLicStr','$voidLicStr')) ";
272  } else
273  {
274  // No filter, nothing to do
275  }
276  $params = array($left, $right, $type, $agentId);
277 
278  $filterParms = $params;
279  $searchFilter = $this->addSearchFilter($filterParms);
280  if (!$activated) {
281  $activatedClause = " AND CE.is_enabled=false ";
282  } else {
283  $activatedClause = " AND cp.".$tableName."_pk NOT IN (SELECT ".$tableName."_fk FROM $tableNameEvent WHERE upload_fk = $upload_pk AND is_enabled = false) ";
284  }
285  $unorderedQuery = "FROM $tableName AS cp " .
286  "INNER JOIN $uploadTreeTableName AS UT ON cp.pfile_fk = UT.pfile_fk " .
287  "LEFT JOIN $tableNameEvent AS CE ON CE.".$tableName."_fk = cp.".$tableName."_pk " .
288  $join .
289  "WHERE cp.content!='' " .
290  "AND ( UT.lft BETWEEN $1 AND $2 ) " .
291  "AND cp.type = $3 " .
292  "AND cp.agent_fk= $4 " .
293  $activatedClause .
294  $sql_upload;
295  $totalFilter = $filterQuery . " " . $searchFilter;
296  $grouping = " GROUP BY cp.content ";
297 
298  $countQuery = "SELECT count(*) FROM (SELECT cp.content, count(*) $unorderedQuery $totalFilter $grouping) as K";
299  $iTotalDisplayRecordsRow = $this->dbManager->getSingleRow($countQuery,
300  $filterParms, __METHOD__.$tableName . ".count" . ($activated ? '' : '_deactivated'));
301  $iTotalDisplayRecords = $iTotalDisplayRecordsRow['count'];
302 
303  $countAllQuery = "SELECT count(*) FROM (SELECT cp.content, count(*) $unorderedQuery$grouping) as K";
304  $iTotalRecordsRow = $this->dbManager->getSingleRow($countAllQuery, $params, __METHOD__,$tableName . "count.all" . ($activated ? '' : '_deactivated'));
305  $iTotalRecords = $iTotalRecordsRow['count'];
306 
307  $range = "";
308  $filterParms[] = $offset;
309  $range .= ' OFFSET $' . count($filterParms);
310  $filterParms[] = $limit;
311  $range .= ' LIMIT $' . count($filterParms);
312 
313  $sql = "SELECT cp.content, cp.hash, CE.content AS contentedited, CE.hash AS hashedited, count(*) AS copyright_count " .
314  $unorderedQuery . $totalFilter . " GROUP BY cp.content, cp.hash, CE.content, CE.hash " . $orderString . $range;
315  $statement = __METHOD__ . $filter.$tableName . $uploadTreeTableName . ($activated ? '' : '_deactivated');
316  $this->dbManager->prepare($statement, $sql);
317  $result = $this->dbManager->execute($statement, $filterParms);
318  $rows = $this->dbManager->fetchAll($result);
319  foreach ($rows as $key => $row) {
320  if (!empty($row['contentedited'])) {
321  $rows[$key]['content'] = $rows[$key]['contentedited'];
322  $row[$key]['hash'] = $rows[$key]['hashedited'];
323  }
324  }
325  $this->dbManager->freeResult($result);
326 
327  return array($rows, $iTotalDisplayRecords, $iTotalRecords);
328  }
