71  protected function handle(Request $request)
72  {
73  /* Is the session owner an admin? */
74  $user_pk = Auth::getUserId();
75  $SessionUserRec = $this->GetUserRec($user_pk);
76  $SessionIsAdmin = $this->IsSessionAdmin($SessionUserRec);
77  $newToken = "";
78 
79  if (GetParm('new_pat', PARM_STRING)) {
80  try {
81  $newToken = $this->generateNewToken($request);
82  } catch (\Exception $e) {
83  $vars['message'] = $e->getMessage();
84  }
85  }
86 
87  $user_pk_to_modify = intval($request->get('user_pk'));
88  if (! ($SessionIsAdmin || empty($user_pk_to_modify) ||
89  $user_pk == $user_pk_to_modify)) {
90  $vars['content'] = _("Your request is not valid.");
91  return $this->render('include/base.html.twig', $this->mergeWithDefault($vars));
92  }
93 
94  $vars = array('refreshUri' => Traceback_uri() . "?mod=" . self::NAME);
95 
96  /*
97  * If this is a POST (the submit button was clicked), then process the
98  * request.
99  */
100  $BtnText = $request->get('UpdateBtn');
101  if (! empty($BtnText)) {
102  /* Get the form data to in an associated array */
103  $UserRec = $this->CreateUserRec($request, "");
104 
105  $rv = $this->UpdateUser($UserRec, $SessionIsAdmin);
106  if (empty($rv)) {
107  // Successful db update
108  $vars['message'] = "User $UserRec[user_name] updated.";
109 
110  /* Reread the user record as update verification */
111  $UserRec = $this->CreateUserRec($request, $UserRec['user_pk']);
112  } else {
113  $vars['message'] = $rv;
114  }
115  } else {
116  $NewUserpk = intval($request->get('newuser'));
117  $UserRec = empty($NewUserpk) ? $this->CreateUserRec($request, $user_pk) : $this->CreateUserRec($request, $NewUserpk);
118  }
119 
120  /* display the edit form with the requested user data */
121  $vars = array_merge($vars, $this->DisplayForm($UserRec, $SessionIsAdmin));
122  $vars['userId'] = $UserRec['user_pk'];
123  $vars['newToken'] = $newToken;
124  $vars['tokenList'] = $this->getListOfActiveTokens();
125  $vars['expiredTokenList'] = $this->getListOfExpiredTokens();
126  $vars['maxTokenDate'] = $this->authHelper->getMaxTokenValidity();

128  $vars['policyRegex'] = generate_password_policy();

130  $vars['formName'] = "user_edit";

202  function UpdateUser($UserRec, $SessionIsAdmin)
203  {
204  global $PG_CONN;
205 
206  $Errors = "";
207 
208  /**** Validations ****/
209  /* Make sure we have a user_pk */
210  if (empty($UserRec['user_pk'])) {
211  $Errors .= "<li>" . _("Consistency error (User_pk missing). Please start over.") . "</li>";
212  }
213 
214  /* Make sure username looks valid */
215  if (empty($UserRec['user_name'])) {
216  $Errors .= "<li>" . _("Username must be specified.") . "</li>";
217  }
218 
219  /* Verify the user_name is not a duplicate */
220  $CheckUserRec = GetSingleRec("users", "WHERE user_name='$UserRec[user_name]'");
221  if ((!empty($CheckUserRec)) and ( $CheckUserRec['user_pk'] != $UserRec['user_pk'])) {
222  $Errors .= "<li>" . _("Username is not unique.") . "</li>";
223  }
224 
225  /* Make sure password matches */
226  if ($UserRec['_pass1'] != $UserRec['_pass2']) {
227  $Errors .= "<li>" . _("Passwords do not match.") . "</li>";
228  }
229 
230  /* Make sure email looks valid */
231  $Check = preg_replace("/[^a-zA-Z0-9@_.+-]/", "", $UserRec['user_email']);
232  if ($Check != $UserRec['user_email']) {
233  $Errors .= "<li>" . _("Invalid email address.") . "</li>";
234  }
235 
236  /* Make sure user can't ask for blank password if policy is enabled */
237  if (passwordPolicyEnabled() && !empty($UserRec['_blank_pass'])) {
238  $Errors .= "<li>" . _("Password policy enabled, can't have a blank password.") . "</li>";
239  }
240 
241  /* Did they specify a password and also request a blank password? */
242  if (!empty($UserRec['_blank_pass']) && ( !empty($UserRec['_pass1']) || ! empty($UserRec['_pass2']))) {
243  $Errors .= "<li>" . _("You cannot specify both a password and a blank password.") . "</li>";
244  }
245 
246  /* Make sure password matches policy */
247  if (!empty($UserRec['_pass1']) && !empty($UserRec['_pass2'])) {
248  $policyRegex = generate_password_policy();
249  $result = preg_match('/^' . $policyRegex . '$/m', $UserRec['_pass1']);
250  if ($result !== 1) {
251  $Errors .= "<li>" . _("Password does not match policy.");
252  $Errors .= "<br />" . generate_password_policy_string();
253  $Errors .= "</li>";
254  }
255  }
256 
257  /* Check if the user is member of the group */
258  if (!empty($UserRec['group_fk'])) {
259  $group_map = $this->userDao->getUserGroupMap($UserRec['user_pk']);
260  if (array_search($UserRec['group_fk'], array_keys($group_map)) === false) {
261  $Errors .= "<li>" . _("User is not member of provided group.") .
262  "</li>";
263  }
264  }
265 
266  /* If we have any errors, return them */
267  if (!empty($Errors)) {
268  return _("Errors") . ":<ol>$Errors </ol>";
269  }
270 
271  /**** Update the users database record ****/
272  /* First remove user_pass and user_seed if the password wasn't changed. */
273  if (!empty($UserRec['_blank_pass']) ) {
274  $UserRec['user_seed'] = '';
275  $options = array('cost' => 10);
276  $UserRec['user_pass'] = password_hash("", PASSWORD_DEFAULT, $options);
277  } else if (empty($UserRec['_pass1'])) { // password wasn't changed
278  unset( $UserRec['user_pass']);
279  unset( $UserRec['user_seed']);
280  }
281 
282  /* Build the sql update */
283  $sql = "UPDATE users SET ";
284  $first = true;
285  foreach ($UserRec as $key=>$val) {
286  if ($key[0] == '_' || $key == "user_pk") {
287  continue;
288  }
289  if (!$SessionIsAdmin && ($key == "user_perm" || $key == "root_folder_fk")) {
290  continue;
291  }
292 
293  if (!$first) {
294  $sql .= ",";
295  }
296  $sql .= "$key='" . pg_escape_string($val) . "'";
297  $first = false;
298  }
299  $sql .= " WHERE user_pk=$UserRec[user_pk]";
300  $result = pg_query($PG_CONN, $sql);
301  DBCheckResult($result, $sql, __FILE__, __LINE__);
302  pg_free_result($result);
303 
304  return (null);
305  } // UpdateUser()

345  function CreateUserRec(Request $request, $user_pk="")
346  {

363  $UserRec['group_fk'] = intval($defaultGroup);

365 
366  $UserRec['_pass1'] = stripslashes($request->get('_pass1'));
367  $UserRec['_pass2'] = stripslashes($request->get('_pass2'));
368  if (!empty($UserRec['_pass1'])) {
369  $UserRec['user_seed'] = 'Seed';
370  $options = array('cost' => 10);
371  $UserRec['user_pass'] = password_hash($UserRec['_pass1'], PASSWORD_DEFAULT, $options);
372  $UserRec['_blank_pass'] = "";
373  } else {
374  $UserRec['user_pass'] = "";
375  $UserRec['_blank_pass'] = stripslashes($request->get("_blank_pass"));
376  if (empty($UserRec['_blank_pass'])) { // check for blank password
377  $StoredUserRec = $this->GetUserRec($UserRec['user_pk']);
378  $options = array('cost' => 10);
379  $UserRec['_blank_pass'] = password_verify($StoredUserRec['user_pass'], password_hash("", PASSWORD_DEFAULT, $options)) ? "on" : "";
380  }
381  }
382 
383  $UserRec['user_perm'] = intval($request->get('user_perm'));
384  $UserRec['user_email'] = stripslashes($request->get('user_email'));
385  $UserRec['email_notify'] = stripslashes($request->get('email_notify'));
386  if (!empty($UserRec['email_notify'])) {
387  $UserRec['email_notify'] = 'y';
388  }
389  $UserRec['user_agent_list'] = userAgents();
390  $UserRec['default_bucketpool_fk'] = intval($request->get("default_bucketpool_fk"));
391  }

405  private function generateNewToken(Request $request)
406  {
407  global $container;
408 
409  $user_pk = Auth::getUserId();
410  $tokenName = GetParm('pat_name', PARM_STRING);
411  $tokenExpiry = GetParm('pat_expiry', PARM_STRING);
412  if ($_SESSION[Auth::USER_LEVEL] < 3) {
413  $tokenScope = 'r';
414  } else {
415  $tokenScope = GetParm('pat_scope', PARM_STRING);
416  }
417  $tokenScope = array_search($tokenScope, RestHelper::SCOPE_DB_MAP);
418  $restHelper = $container->get('helper.restHelper');
419  $isTokenRequestValid = $restHelper->validateTokenRequest($tokenExpiry,
420  $tokenName, $tokenScope);
421 
422  if ($isTokenRequestValid !== true) {
423  throw new \UnexpectedValueException($isTokenRequestValid->getMessage());
424  } else {
425  $restDbHelper = $container->get('helper.dbHelper');
426  $key = bin2hex(
427  openssl_random_pseudo_bytes(RestHelper::TOKEN_KEY_LENGTH / 2));
428  try {
429  $jti = $restDbHelper->insertNewTokenKey($user_pk, $tokenExpiry,
430  RestHelper::SCOPE_DB_MAP[$tokenScope], $tokenName, $key);
431  } catch (DuplicateTokenKeyException $e) {
432  // Key already exists, try again.
433  $key = bin2hex(
434  openssl_random_pseudo_bytes(RestHelper::TOKEN_KEY_LENGTH / 2));
435  try {
436  $jti = $restDbHelper->insertNewTokenKey($user_pk, $tokenExpiry,
437  RestHelper::SCOPE_DB_MAP[$tokenScope], $tokenName, $key);
438  } catch (DuplicateTokenKeyException $e) {
439  // New key also failed, give up!
440  throw new DuplicateTokenKeyException("Please try again later.");
441  }
442  } catch (DuplicateTokenNameException $e) {
443  throw new \UnexpectedValueException($e->getMessage());
444  }
445  return $this->authHelper->generateJwtToken($tokenExpiry,
446  $jti['created_on'], $jti['jti'], $tokenScope, $key);
447  }
