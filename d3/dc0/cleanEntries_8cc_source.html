40 {
41  stringstream ss;

43  string s = ss.str();
44  string::size_type len = s.length();

46  {
47  char cBegin = s[0];
48  char cEnd = s[len - 1];
49  if (cBegin == ' ' && cEnd == ' ')

52  return s.substr(1);
53  else if (cEnd == ' ')
54  return s.substr(0, len - 1);
