97  public function getUploads($request, $response, $args)
98  {
99  $id = null;
100  $folderId = null;
101  $recursive = true;
102  $retVal = null;
103  $query = $request->getQueryParams();
104 
105  if (array_key_exists(self::FOLDER_PARAM, $query)) {
106  $folderId = filter_var($query[self::FOLDER_PARAM], FILTER_VALIDATE_INT);
107  if (! $this->restHelper->getFolderDao()->isFolderAccessible($folderId,
108  $this->restHelper->getUserId())) {
109  $info = new Info(404, "Folder does not exist", InfoType::ERROR);
110  $retVal = $response->withJson($info->getArray(), $info->getCode());
111  }
112  }
113 
114  if (array_key_exists(self::RECURSIVE_PARAM, $query)) {
115  $recursive = filter_var($query[self::RECURSIVE_PARAM],
116  FILTER_VALIDATE_BOOLEAN);
117  }
118 
119  $page = $request->getHeaderLine(self::PAGE_PARAM);
120  if (! empty($page)) {
121  $page = filter_var($page, FILTER_VALIDATE_INT);
122  if ($page <= 0) {
123  $info = new Info(400, "page should be positive integer > 0",
124  InfoType::ERROR);
125  $retVal = $response->withJson($info->getArray(), $info->getCode());
126  }
127  } else {
128  $page = 1;
129  }
130 
131  $limit = $request->getHeaderLine(self::LIMIT_PARAM);
132  if (! empty($limit)) {
133  $limit = filter_var($limit, FILTER_VALIDATE_INT);
134  if ($limit < 1) {
135  $info = new Info(400, "limit should be positive integer > 1",
136  InfoType::ERROR);
137  $retVal = $response->withJson($info->getArray(), $info->getCode());
138  }
139  } else {
140  $limit = self::UPLOAD_FETCH_LIMIT;
141  }
142 
143  if (isset($args['id'])) {
144  $id = intval($args['id']);
145  $upload = $this->uploadAccessible($this->restHelper->getGroupId(), $id);
146  if ($upload !== true) {
147  return $response->withJson($upload->getArray(), $upload->getCode());
148  }
149  $temp = $this->isAdj2nestDone($id, $response);
150  if ($temp !== true) {
151  $retVal = $temp;
152  }
153  }
154  if ($retVal !== null) {
155  return $retVal;
156  }
157  list($pages, $uploads) = $this->dbHelper->getUploads(
158  $this->restHelper->getUserId(), $this->restHelper->getGroupId(), $limit,
159  $page, $id, $folderId, $recursive);
160  if ($id !== null && ! empty($uploads)) {
161  $uploads = $uploads[0];
162  $pages = 1;
163  }
164  return $response->withHeader("X-Total-Pages", $pages)->withJson($uploads,
165  200);
166  }

414  private function areAgentsScheduled($uploadId, $agents, $response)
415  {
416  global $container;
417  $agentDao = $container->get('dao.agent');
418 
419  $agentList = array_keys(AgentRef::AGENT_LIST);
420  $intersectArray = array_intersect($agents, $agentList);
421 
422  $error = null;
423  if (count($agents) != count($intersectArray)) {
424  $error = new Info(400, "Agent should be any of " .
425  implode(", ", $agentList) . ". " . implode(",", $agents) . " passed.",
426  InfoType::ERROR);
427  } else {
428  // Agent is valid, check if they have ars tables.
429  foreach ($agents as $agent) {
430  if (! $agentDao->arsTableExists($agent)) {
431  $error = new Info(412, "Agent $agent not scheduled for the upload. " .
432  "Please POST to /jobs", InfoType::ERROR);
433  break;
434  }
435  }
436  }
437  if ($error !== null) {
438  return $response->withJson($error->getArray(), $error->getCode());
439  }
440 
441  $scanProx = new ScanJobProxy($agentDao, $uploadId);
442  $agentList = $scanProx->createAgentStatus($agents);
443 
444  foreach ($agentList as $agent) {
445  if (! array_key_exists('currentAgentId', $agent)) {
446  $error = new Info(412, "Agent " . $agent["agentName"] .
447  " not scheduled for the upload. Please POST to /jobs",
448  InfoType::ERROR);
449  $response = $response->withJson($error->getArray(), $error->getCode());
450  } else if (array_key_exists('isAgentRunning', $agent) &&
451  $agent['isAgentRunning']) {
452  $error = new Info(503, "Agent $agent is running. " .
453  "Please check job status at /api/v1/jobs?upload=" . $uploadId,
454  InfoType::INFO);
455  $response = $response->withHeader('Retry-After', '60')
456  ->withHeader('Look-at', "/api/v1/jobs?upload=" . $uploadId)
457  ->withJson($error->getArray(), $error->getCode());
458  }
459  if ($error !== null) {
460  return $response;
461  }
462  }
463  return true;
464  }
