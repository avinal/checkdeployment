44  function Add()
45  {
46 
47  global $PG_CONN;
48 
49  if (! $PG_CONN) {
50  DBconnect();
51  if (! $PG_CONN) {
52  $text = _("NO DB connection!");
53  echo "<pre>$text\n</pre>";
54  }
55  }
56 
57  /* Get the parameters */
58  $User = str_replace("'", "''", GetParm('username', PARM_TEXT));
59  $User = trim($User);
60  $Pass = GetParm('pass1', PARM_TEXT);
61  $Pass2 = GetParm('pass2', PARM_TEXT);
62  $options = array('cost' => 10);
63  $Hash = password_hash($Pass, PASSWORD_DEFAULT, $options);
64  $Desc = str_replace("'", "''", GetParm('description', PARM_TEXT));
65  $Perm = GetParm('permission', PARM_INTEGER);
66  $Folder = GetParm('folder', PARM_INTEGER);
67  $Email_notify = GetParm('enote', PARM_TEXT);
68  $Email = str_replace("'", "''", GetParm('email', PARM_TEXT));
69  $agentList = userAgents();
70  $default_bucketpool_fk = GetParm('default_bucketpool_fk', PARM_INTEGER);
71 
72  /* Make sure username looks valid */
73  if (empty($User)) {
74  $text = _("Username must be specified. Not added.");
75  return ($text);
76  }
77  /* limit the user name size to 64 characters when creating an account */
78  if (strlen($User) > 64) {
79  $text = _("Username exceed 64 characters. Not added.");
80  return ($text);
81  }
82  /* Make sure password matches */
83  if ($Pass != $Pass2) {
84  $text = _("Passwords did not match. Not added.");
85  return ($text);
86  }
87 
88  /* Make sure password matches policy */
89  $policyRegex = generate_password_policy();
90  $result = preg_match('/^' . $policyRegex . '$/m', $Pass);
91  if ($result !== 1) {
92  $text = _("Password does not match policy.");
93  $text .= "<br />" . generate_password_policy_string();
94  return ($text);
95  }
96 
97  if (empty($Email)) {
98  $text = _("Email must be specified. Not added.");
99  return ($text);
100  }
101 
102  /* Make sure email looks valid */
103  if (! filter_var($Email, FILTER_VALIDATE_EMAIL)) {
104  $text = _("Invalid email address. Not added.");
105  return ($text);
106  }
107 
108  /* Make sure email is unique */
109  $email_count = $this->dbManager->getSingleRow(
110  "SELECT COUNT(*) as count FROM users WHERE user_email = $1 LIMIT 1;",
111  array($Email))["count"];
112  if ($email_count > 0) {
113  $text = _("Email address already exists. Not added.");
114  return ($text);
115  }
116 
117  /* See if the user already exists (better not!) */
118  $row = $this->dbManager->getSingleRow("SELECT * FROM users WHERE LOWER(user_name) = LOWER($1) LIMIT 1;",
119  array($User), $stmt = __METHOD__ . ".getUserIfExisting");
120  if (! empty($row['user_name'])) {
121  $text = _("User already exists. Not added.");
122  return ($text);
123  }
124 
125  /* check email notification, if empty (box not checked), or if no email
126  * specified for the user set to 'n'.
127  */
128  if (empty($Email_notify) || empty($Email)) {
129  $Email_notify = '';
130  }
131 
132  $ErrMsg = add_user($User, $Desc, $Hash, $Perm, $Email, $Email_notify,
133  $agentList, $Folder, $default_bucketpool_fk);
134 
135  return ($ErrMsg);
136  } // Add()
