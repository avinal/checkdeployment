40  public function getFullPath($itemId, $tableName, $parentId=0, $dropArtifactPrefix=false)
41  {
42  $statementName = __METHOD__.".".$tableName;
43 
44  if ($parentId==$itemId) {
45  return $this->getFullPath($itemId, $tableName);
46  } else if ($parentId > 0) {
47  $params = array($itemId, $parentId);
48  $parentClause = " = $2";
49  $parentLoopCondition = "AND (ut.parent != $2)";
50  $statementName .= ".parent";
51  } else {
52  $params = array($itemId);
53  $parentClause = " IS NULL";
54  $parentLoopCondition = "";
55  }
56 
57  $row = $this->dbManager->getSingleRow(
58  $sql= "
59  WITH RECURSIVE file_tree(uploadtree_pk, parent, ufile_name, path, prev_ufile_mode, artifact_path_prefix, file_path, cycle) AS (
60  SELECT ut.uploadtree_pk, ut.parent, ut.ufile_name,
61  ARRAY[ut.uploadtree_pk],
62  ut.ufile_mode,
63  '',
64  CASE WHEN ut.ufile_mode & (1<<28) = 0 THEN ut.ufile_name ELSE '' END,
65  false
66  FROM $tableName ut
67  WHERE ut.uploadtree_pk = $1
68  UNION ALL
69  SELECT ut.uploadtree_pk, ut.parent, ut.ufile_name,
70  path || ut.uploadtree_pk,
71  ut.ufile_mode,
72  CASE WHEN prev_ufile_mode & (1<<28) = 0
73  THEN
74  CASE WHEN ut.ufile_mode & (1<<28) = 0
75  THEN ''
76  ELSE artifact_path_prefix
77  END
78  ELSE
79  CASE WHEN ut.ufile_mode & (1<<28) = 0
80  THEN ut.ufile_name || '/' || artifact_path_prefix
81  ELSE artifact_path_prefix
82  END
83  END,
84  CASE WHEN (prev_ufile_mode & (1<<28) = 0 and ut.ufile_mode & (1<<28) = 0)
85  THEN ut.ufile_name || '/' || artifact_path_prefix || file_path
86  ELSE file_path
87  END,
88  (ut.uploadtree_pk = ANY(path)) $parentLoopCondition
89  FROM $tableName ut, file_tree ft
90  WHERE ut.uploadtree_pk = ft.parent AND NOT cycle
91  )
92  SELECT artifact_path_prefix, file_path from file_tree WHERE parent $parentClause",
93  $params, $statementName);
94 
95  if (false === $row) {
96  throw new \Exception("could not find path of $itemId:\n$sql--".print_r($params,true));
97  }
98 
99  if (! $dropArtifactPrefix) {
100  return $row['artifact_path_prefix'].$row['file_path'];
101  } else {
102  return $row['file_path'];
103  }
104  }

106  public function getMinimalCoveringItem($uploadId, $tableName)
107  {
108  $statementName = __METHOD__.".".$tableName;
109 
110  $row = $this->dbManager->getSingleRow(
111  "SELECT uploadtree_pk FROM $tableName ut WHERE ut.upload_fk = $1
112  AND NOT EXISTS (
113  SELECT 1 FROM $tableName ut2 WHERE ut2.upload_fk = $1
114  AND NOT (ut2.lft BETWEEN ut.lft AND ut.rgt)
115  AND (ut2.ufile_mode & (3<<28) = 0)
116  )
117  ORDER BY ut.lft DESC LIMIT 1",
118  array($uploadId),
119  $statementName
120  );
121 
122  return $row ? $row['uploadtree_pk'] : 0;
123  }

137  {
138  $pfileRow = $this->dbManager->getSingleRow('SELECT * FROM pfile WHERE pfile_pk=$1',array($pfileId));
139  global $LIBEXECDIR;
140  if (empty($pfileRow['pfile_sha1'])) {
141  return null;

143  $hash = $pfileRow['pfile_sha1'] . "." . $pfileRow['pfile_md5'] . "." . $pfileRow['pfile_size'];
144  $path = '';
