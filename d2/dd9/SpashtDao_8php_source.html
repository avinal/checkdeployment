145  public function updateCopyright($item, $hash, $content, $action='')
146  {
147  $itemTable = $item->getUploadTreeTableName();
148  $stmt = __METHOD__ . "$itemTable";
149  $params = array($hash, $item->getLeft(), $item->getRight());
150 
151  if ($action == "delete") {
152  $setSql = "is_enabled='false'";
153  $stmt .= '.delete';
154  } else if ($action == "rollback") {
155  $setSql = "is_enabled='true'";
156  $stmt .= '.rollback';
157  } else {
158  $setSql = "textfinding = $4, hash = md5($4), is_enabled='true'";
159  $params[] = StringOperation::replaceUnicodeControlChar($content);
160  }
161 
162  $sql = "UPDATE copyright_spasht AS cpr SET $setSql
163  FROM copyright_spasht as cp
164  INNER JOIN $itemTable AS ut ON cp.pfile_fk = ut.pfile_fk
165  WHERE cpr.copyright_spasht_pk = cp.copyright_spasht_pk
166  AND cp.hash = $1
167  AND ( ut.lft BETWEEN $2 AND $3 )";
168  if ('uploadtree_a' == $item->getUploadTreeTableName()) {
169  $params[] = $item->getUploadId();
170  $sql .= " AND ut.upload_fk=$".count($params);
171  $stmt .= '.upload';
172  }
173  $this->dbManager->getSingleRow($sql, $params, $stmt);
174  }
