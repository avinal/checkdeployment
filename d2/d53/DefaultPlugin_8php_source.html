290  protected function render($templateName, $vars = null, $headers = null)
291  {
292  if ($this->requiresLogin && !$this->isLoggedIn()) {
293  new Response("permission denied", Response::HTTP_FORBIDDEN, array("contentType" => "text/plain"));
