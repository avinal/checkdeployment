119  """

137  def get_diff_dir(self):
138  """
139  Populate temp dir using the gitlab API `merge_requests`
140 
141  :return: temp dir path
142  :rtype: string
143  """
144  change_response = None
145  path_key = None
146  change_key = None
147 
148  if RUNNING_ON == "GITLAB":
149  api_req_url = f"{API_URL}/projects/{PROJECT_ID}/merge_requests" + \
150  f"/{MR_IID}/changes"
151  headers = {'Private-Token': API_TOKEN}
152  path_key = "new_path"
153  change_key = "diff"
154  else:
155  api_req_url = f"https://api.github.com/repos/{TRAVIS_REPO_SLUG}" + \
156  f"/pulls/{TRAVIS_PULL_REQUEST}/files"
157  headers = {}
158  path_key = "filename"
159  change_key = "patch"
160 
161  context = ssl.create_default_context()
162  context.check_hostname = False
163  context.verify_mode = ssl.CERT_NONE
164 
165  req = urllib.request.Request(api_req_url, headers=headers)
166  try:
167  with urllib.request.urlopen(req, context=context) as response:
168  change_response = response.read()
169  except Exception as e:
170  print(f"Unable to get URL {api_req_url}")
171  raise e
172 
173  change_response = json.loads(change_response)
174  if RUNNING_ON == "GITLAB":
175  changes = change_response['changes']
176  else:
177  changes = change_response
178 
179  remove_diff_regex = re.compile(r"^([ +-])(.*)$", re.MULTILINE)
180 
181  for change in changes:
182  if change[path_key] is not None:
183  path_to_be_excluded = self.__is_excluded_path(change[path_key])
184  if path_to_be_excluded == False:
185  curr_file = os.path.join(self.temp_dir.name, change[path_key])
186  curr_dir = os.path.dirname(curr_file)
187  if curr_dir != self.temp_dir.name:
188  os.makedirs(name=curr_dir, exist_ok=True)
189  curr_file = open(file=curr_file, mode='w+', encoding='UTF-8')
190  print(re.sub(remove_diff_regex, r"\2", change[change_key]),
191  file=curr_file)
192 
193  return self.temp_dir.name

413  def __merge_nomos_ojo(self, nomos_licenses, ojo_licenses):
414  """
415  Merge the results from nomos and ojo based on file name
416 
417  :param nomos_licenses: formatted result form nomos

420  :type ojo_licenses: list()
421 
422  :return: merged list of scanner findings
423  :rtype: list()
424  """
425  for ojo_entry in ojo_licenses:
426  for nomos_entry in nomos_licenses:
427  if ojo_entry['file'] == nomos_entry['file']:
428  nomos_entry['result'].update(ojo_entry['result'])
429  break
430  else:
431  nomos_licenses.append(ojo_entry)

434  def results_are_whitelisted(self):
435  """
436  Get the formatted list of license scanner findings

440 
441  :return: merged list of scanner findings
442  :rtype: list()
443  """

446  nomos_licenses = self.__get_non_whitelisted_license_nomos()
447  if self.cli_options.ojo == False:
448  failed_licenses = nomos_licenses

454  failed_licenses = self.__merge_nomos_ojo(nomos_licenses,
455  ojo_licenses)
456  if len(failed_licenses) > 0:
457  return failed_licenses
458  return True
