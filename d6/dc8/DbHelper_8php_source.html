98  public function getUploads($userId, $groupId, $limit, $page = 1,
99  $uploadId = null, $folderId = null, $recursive = true)
100  {
101  $uploadProxy = new UploadBrowseProxy($groupId, 0, $this->dbManager);
102  if ($folderId === null) {
103  $user = $this->getUsers($userId);
104  $folderId = $user[0]['rootFolderId'];
105  }
106  $folders = [$folderId];
107 
108  if ($uploadId !== null) {
109  $recursive = true;
110  $user = $this->getUsers($userId);
111  $folderId = $user[0]['rootFolderId'];
112  $folders = [$folderId];
113  }
114 
115  if ($recursive) {
116  $tree = $this->folderDao->getFolderStructure($folderId);
117  $folders = array_map(function ($folder) {
118  return $folder[$this->folderDao::FOLDER_KEY]->getId();
119  }, $tree);
120  }
121 
122  $params = [$folders];
123  $partialQuery = $uploadProxy->getFolderPartialQuery($params);
124 
125  $where = "";
126  $statementCount = __METHOD__ . ".countAllUploads";
127  $statementGet = __METHOD__ . ".getAllUploads.$limit";
128  if ($uploadId !== null) {
129  $params[] = $uploadId;
130  $where = "AND upload.upload_pk = $" . count($params);
131  $statementGet .= ".upload";
132  $statementCount .= ".upload";
133  }
134  $sql = "SELECT count(*) AS cnt FROM $partialQuery $where;";
135  $totalResult = $this->dbManager->getSingleRow($sql, $params, $statementCount);
136  $totalResult = intval($totalResult['cnt']);
137  $totalResult = intval(floor($totalResult / $limit) + 1);
138 
139  $params[] = ($page - 1) * $limit;
140 
141  $sql = "SELECT
142 upload.upload_pk, upload.upload_desc, upload.upload_ts, upload.upload_filename
143 FROM $partialQuery $where ORDER BY upload_pk ASC LIMIT $limit OFFSET $" .
144  count($params) . ";";
145  $results = $this->dbManager->getRows($sql, $params, $statementGet);
146  $uploads = [];
147  foreach ($results as $row) {
148  $uploadId = $row["upload_pk"];
149  $pfile_size = null;
150  $pfile_sha1 = null;
151  $pfile_md5 = null;
152  $pfile_sha256 = null;
153  $pfile = $this->getPfileInfoForUpload($uploadId);
154  if ($pfile !== null) {
155  $pfile_size = $pfile['pfile_size'];
156  $pfile_sha1 = $pfile['pfile_sha1'];
157  $pfile_md5 = $pfile['pfile_md5'];
158  $pfile_sha256 = $pfile['pfile_sha256'];
159  }
160 
161  $folder = $this->getFolderForUpload($uploadId);
162  if ($folder === null) {
163  continue;
164  }
165  $folderId = $folder->getId();
166  $folderName = $folder->getName();
167 
168  $hash = new Hash($pfile_sha1, $pfile_md5, $pfile_sha256, $pfile_size);
169  $upload = new Upload($folderId, $folderName, $uploadId,
170  $row["upload_desc"], $row["upload_filename"], $row["upload_ts"], $hash);
171  array_push($uploads, $upload->getArray());
172  }
173  return [$totalResult, $uploads];
174  }

251  null, null, null, null, null);
