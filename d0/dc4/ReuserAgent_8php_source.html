105  function processUploadId($uploadId)
106  {
107  $itemTreeBounds = $this->uploadDao->getParentItemBounds($uploadId);
108  foreach ($this->uploadDao->getReusedUpload($uploadId, $this->groupId) as $reuseTriple) {
109  // Get the reuse upload id
110  $reusedUploadId = $reuseTriple['reused_upload_fk'];
111  // Get the group id
112  $reusedGroupId = $reuseTriple['reused_group_fk'];
113  // Get the reuse mode
114  $reuseMode = $reuseTriple['reuse_mode'];
115  // Get the ItemTreeBounds for the upload
116  $itemTreeBoundsReused = $this->uploadDao->getParentItemBounds($reusedUploadId);
117  if (false === $itemTreeBoundsReused) {
118  continue;
119  }
120 
121  if ($reuseMode & UploadDao::REUSE_ENHANCED) {
122  $this->processEnhancedUploadReuse($itemTreeBounds, $itemTreeBoundsReused, $reusedGroupId);
123  } else {
124  $this->processUploadReuse($itemTreeBounds, $itemTreeBoundsReused, $reusedGroupId);
125  }
126 
127  if ($reuseMode & UploadDao::REUSE_MAIN) {
128  $this->reuseMainLicense($uploadId, $this->groupId, $reusedUploadId, $reusedGroupId);
129  }
130 
131  if ($reuseMode & UploadDao::REUSE_CONF) {
132  $this->reuseConfSettings($uploadId, $reusedUploadId);
133  }
134 
135  if ($reuseMode & UploadDao::REUSE_COPYRIGHT) {
136  $this->reuseCopyrights($uploadId, $reusedUploadId);
137  }
138  }
139  return true;
140  }

207  protected function reuseCopyrights($uploadId, $reusedUploadId)
208  {
209  $agentId = $this->getAgentId($uploadId);
210  $reusedAgentId = $this->getAgentId($reusedUploadId);
211  if ($agentId == null || $reusedAgentId == null) {
212  return true;
213  }
214  $uploadTreeTableName = $this->uploadDao->getUploadtreeTableName($uploadId);
215  $extrawhere = ' agent_fk='.$agentId;
216  $allCopyrights = $this->copyrightDao->getScannerEntries('copyright', $uploadTreeTableName, $uploadId, $type=null,
217  $extrawhere, $enabled='true');
218 
219  $reusedCopyrights = $this->copyrightDao->getAllEventEntriesForUpload($reusedUploadId, $reusedAgentId);
220 
221  if (!empty($reusedCopyrights) && !empty($allCopyrights)) {
222  foreach ($reusedCopyrights as $reusedCopyright) {
223  foreach ($allCopyrights as $copyright) {
224  if (strcmp($copyright['hash'], $reusedCopyright['hash']) == 0) {
225  $action = "delete";
226  $content = "";
227  $hash = "";
228  if ($reusedCopyright['is_enabled']) {
229  $action = "";
230  $content = $reusedCopyright['content'];
231  $hash = $copyright['hash'];
232  }
233  $item = $this->uploadDao->getItemTreeBounds(intval($copyright['uploadtree_pk']),
234  $uploadTreeTableName);
235  $this->copyrightDao->updateTable($item, $hash, $content, $this->userId,
236  'copyright', $action);
237  $this->heartbeat(1);
238  }
239  }
240  }
241  }
242  return true;

253  protected function processUploadReuse($itemTreeBounds, $itemTreeBoundsReused, $reusedGroupId)
254  {
255  $groupId = $this->groupId;

257 
258  $clearingDecisions = $this->clearingDao->getFileClearingsFolder($itemTreeBoundsReused, $reusedGroupId);

260 
261  $currenlyVisibleClearingDecisionsById = $this->mapByClearingId($currenlyVisibleClearingDecisions);
262  $clearingDecisionsById = $this->mapByClearingId($clearingDecisions);
263 
264  $clearingDecisionsToImport = array_diff_key($clearingDecisionsById,$currenlyVisibleClearingDecisionsById);
265 
266  $clearingDecisionToImportByFileId = $this->mapByFileId($clearingDecisionsToImport);
267 
268  $uploadDao = $this->uploadDao;
270  $containedItems = ArrayOperation::callChunked(
271  function ($fileIds) use ($itemTreeBounds, $uploadDao)
272  {
273  return $uploadDao->getContainedItems(
274  $itemTreeBounds,
275  "pfile_fk = ANY($1)",
276  array('{' . implode(', ', $fileIds) . '}')
277  );
278  }, array_keys($clearingDecisionToImportByFileId), 100);
279 
280  foreach ($containedItems as $item) {
281  $fileId = $item->getFileId();
282  if (empty($fileId)) {
283  continue;
284  }
285  if (array_key_exists($fileId, $clearingDecisionToImportByFileId)) {
286  $this->createCopyOfClearingDecision($item->getId(), $userId, $groupId,
287  $clearingDecisionToImportByFileId[$fileId]);
288  } else {
289  throw new \Exception("bad internal state");
290  }
291 
292  $this->heartbeat(1);
293  }
294 
295  return true;
