142  function updateCommentFromArray($commentArray)

146  return false;
147  }
148 
149  $userFk = Auth::getUserId();
150  $updated = 0;
151 
152  foreach ($commentArray as $commentPk => $comment) {
153  if (count($comment) < 1 ||
154  (! array_key_exists("name", $comment) &&
155  ! array_key_exists("comment", $comment))) {
156  throw new \UnexpectedValueException(
157  "At least name or comment is " . "required for entry " . $commentPk);
158  }
159  $this->isCommentIdValid($commentPk);
160  $statement = __METHOD__;
161  $params = [$commentPk, $userFk];
162  $updateStatement = [];
163  if (array_key_exists("name", $comment)) {
164  $params[] = $comment["name"];
165  $updateStatement[] = "name = $" . count($params);
166  $statement .= ".name";
167  }
168  if (array_key_exists("comment", $comment)) {
169  $params[] = StringOperation::replaceUnicodeControlChar($comment["comment"]);
170  $updateStatement[] = "comment = $" . count($params);
171  $statement .= ".comment";
172  }
173  $sql = "UPDATE license_std_comment " .
174  "SET updated = NOW(), user_fk = $2, " . join(",", $updateStatement) .
175  "WHERE lsc_pk = $1 " .
176  "RETURNING 1 AS updated;";
177  $retVal = $this->dbManager->getSingleRow($sql, $params, $statement);
178  $updated += intval($retVal);
179  }
180  return $updated;
