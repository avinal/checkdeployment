62  protected function handleUpload(Request $request)
63  {
64  global $MODDIR;
65  global $SYSCONFDIR;
66  global $Plugins;
67 
68  $folderId = intval($request->get(self::FOLDER_PARAMETER_NAME));
69  $description = stripslashes($request->get(self::DESCRIPTION_INPUT_NAME));
70  $description = $this->basicShEscaping($description);
71 
72  $getUrlThatMightIncludeSpaces = trim($request->get(self::GETURL_PARAM));
73  $getUrl = str_replace(" ", "%20", $getUrlThatMightIncludeSpaces);
74 
75  if (empty($getUrl)) {
76  return array(false, _("Empty URL") . $getUrl, $description);
77  }
78  if (preg_match("@^((http)|(https))://([[:alnum:]]+)@i", $getUrl) != 1) {
79  return array(false, _("Invalid URL") . $getUrl, $description);
80  }
81  $getUrl = $this->basicShEscaping($getUrl);
82 
83  if ($request->getSession()->get(self::UPLOAD_FORM_BUILD_PARAMETER_NAME)
84  != $request->get(self::UPLOAD_FORM_BUILD_PARAMETER_NAME)) {
85  $text = _("This seems to be a resent file.");
86  return array(false, $text, $description);
87  }
88 
89  if (empty($folderId)) {
90  $text = _("Invalid Folder.");
91  return array(false, $text, $description);
92  }
93 
94  $public = $request->get('public');
95  $publicPermission = ($public == self::PUBLIC_ALL) ? Auth::PERM_READ : Auth::PERM_NONE;
96 
97  $Name = trim($request->get('name'));
98  if (empty($Name)) {
99  $Name = basename($getUrl);
100  }
101  $ShortName = basename($Name);
102  if (empty($ShortName)) {
103  $ShortName = $Name;
104  }
105 
106  /* Create an upload record. */
107  $uploadMode = (1 << 2); // code for "it came from wget"
108  $userId = Auth::getUserId();
109  $groupId = Auth::getGroupId();
110  $uploadId = JobAddUpload($userId, $groupId, $ShortName, $getUrl,
111  $description, $uploadMode, $folderId, $publicPermission);
112  if (empty($uploadId)) {
113  $text = _("Failed to insert upload record");
114  return array(false, $text, $description);
115  }
116 
117  /* Create the job: job "wget" */
118  $jobpk = JobAddJob($userId, $groupId, "wget", $uploadId);
119  if (empty($jobpk) || ($jobpk < 0)) {
120  $text = _("Failed to insert job record");
121  return array(false, $text, $description);
122  }
123 
124  $VCSType = trim($request->get('vcstype'));
125  $VCSType = $this->basicShEscaping($VCSType);
126  $jq_args = "$uploadId - $getUrl $VCSType ";
127 
128  $Username = trim($request->get('username'));
129  $Username = $this->basicShEscaping($Username);
130  if (!empty($Username)) {
131  $jq_args .= "--username $Username ";
132  }
133 
134  $Passwd = trim($request->get('passwd'));
135  $Passwd = $this->basicShEscaping($Passwd);
136  if (!empty($Passwd)) {
137  $jq_args .= "--password $Passwd ";
138  }
139 
140  $Branch = trim(explode(' ',trim($request->get('branch')))[0]);
141  if (!empty($Branch) && strcasecmp($VCSType,'git') == 0) {
142  $jq_args .= "--single-branch --branch '$Branch'";
143  }
144 
145  $jobqueuepk = JobQueueAdd($jobpk, "wget_agent", $jq_args, NULL, NULL);
146  if (empty($jobqueuepk)) {
147  $text = _("Failed to insert task 'wget_agent' into job queue");
148  return array(false, $text, $description);
149  }
150  /* schedule agents */
151  $unpackplugin = &$Plugins[plugin_find_id("agent_unpack") ];
152  $unpackArgs = intval($request->get('scm')) == 1 ? '-I' : '';
153  $ununpack_jq_pk = $unpackplugin->AgentAdd($jobpk, $uploadId, $ErrorMsg, array("wget_agent"), $unpackArgs);
154  if ($ununpack_jq_pk < 0) {
155  return array(false, _($ErrorMsg), $description);
156  }
157 
158  $adj2nestplugin = &$Plugins[plugin_find_id("agent_adj2nest") ];
159  $adj2nest_jq_pk = $adj2nestplugin->AgentAdd($jobpk, $uploadId, $ErrorMsg, array());
160  if ($adj2nest_jq_pk < 0) {
161  return array(false, _($ErrorMsg), $description);
162  }
163 
164  AgentCheckBoxDo($jobpk, $uploadId);
165 
166  $msg = "";
168  $status = GetRunnableJobList();
169  if (empty($status)) {
170  $msg .= _("Is the scheduler running? ");
171  }
172  $Url = Traceback_uri() . "?mod=showjobs&upload=$uploadId";
173  $text = _("The upload");
174  $text1 = _("has been queued. It is");
175  $msg .= "$text $Name $text1 ";
176  $keep = "<a href='$Url'>upload #" . $uploadId . "</a>.\n";
177  return array(true, $msg.$keep, $description, $uploadId);
178  }
