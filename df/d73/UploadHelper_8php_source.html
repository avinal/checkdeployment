194  private function handleUpload($body, $uploadType, $folderId, $fileDescription,
195  $isPublic, $ignoreScm = 0)
196  {
197  $sanity = false;
198  switch ($uploadType) {

202  case "url":
203  $sanity = $this->sanitizeUrlData($body);
204  break;
205  case "server":
206  $sanity = $this->sanitizeSrvData($body);
207  break;
208  default:
209  $message = "Invalid 'uploadType'";

214  }
215  if ($sanity !== true) {
216  return $sanity;
217  }
218  $uploadResponse = false;
219  switch ($uploadType) {
220  case "vcs":
221  $uploadResponse = $this->generateVcsUpload($body, $folderId,
